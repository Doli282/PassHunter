# public

/index  ... [GET]   ... homepage

# authentication

/login          ... [GET, POST]
/logout         ... [POST] ([GET])
/register       ... [GET, POST] ... register new account - sef-service
/reset-password ... [GET, POST] ... reset password - self-service

# Domain search

/search? ... [GET/POST] ... search domain in database (POST because of the length of possible query)

# watchlist management

/watchlist                                                  ... [GET]       ... get list of watchlists
/watchlist/create-watchlist                                 ... [GET, POST] ... create new watchlist
/watchlist/<watchlist_id>                                   ... [GET]       ... get details of a watchlist
/watchlist/<watchlist_id>/update-watchlist                  ... [GET, POST] ... update watchlist information
/watchlist/<watchlist_id>/deactivate-watchlist              ... [POST]      ... deactivate watchlist
/watchlist/<watchlist_id>/activate-watchlist                ... [POST]      ... activate watchlist
/watchlist/<watchlist_id>/delete-watchlist                  ... [GET, POST] ... delete a watchlist
/watchlist/<watchlist_id>/add-domain                        ... [GET, POST] ... add new domain to a watchlist
/watchlist/<watchlist_id>/domain/<domain_id>/remove-domain  ... [GET, POST] ... remove a domain from a watchlist

# alerts

/alert                          ... [GET]   ... get list of alerts
/alert/<alert_id>               ... [GET]   ... get details of an alert
/alert/<alert_id>/register      ... [POST]  ... register alert
/alert/<alert_id>/deregister    ... [POST]  ... deregister alert

# profile

/profile                    ... [GET]       ... get profile information
/profile/change-password    ... [GET, POST] ... change password - self-service
/profile/update             ... [GET, POST] ... change profile details - self-service
/profile/delete             ... [GET, POST] ... delete profile - self-service

# Admin - user management

/user   ... [GET] ...   get list of users
/user/<user_id> ... [GET] ... get user details
/user/<user_id>/update ... [GET] ([POST]) ... update user
/user/<user_id>/deactivate ... [POST] ... deactivate user
/user/<user_id>/activate ... [POST] ... activate user
